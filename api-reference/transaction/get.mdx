---
title: Get transaction
description: Retrieve details of a specific transaction
api: GET /api/external/transaction/{id}
---

Retrieve detailed information about a specific transaction using its unique identifier.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the transaction
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The transaction details
  
  <Expandable title="Transaction object">
    <ResponseField name="id" type="string">
      Unique identifier for the transaction
    </ResponseField>
    
    <ResponseField name="type" type="string">
      Transaction type (payout, collection)
    </ResponseField>
    
    <ResponseField name="customer_id" type="string">
      The customer ID associated with the transaction
    </ResponseField>
    
    <ResponseField name="wallet_id" type="string">
      The wallet ID involved in the transaction
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      The transaction amount
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency code
    </ResponseField>
    
    <ResponseField name="method" type="string">
      The payment method used
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Transaction status (pending, processing, completed, failed, cancelled)
    </ResponseField>
    
    <ResponseField name="reference" type="string">
      Your unique reference for the transaction
    </ResponseField>
    
    <ResponseField name="narration" type="string">
      Description or note for the transaction
    </ResponseField>
    
    <ResponseField name="fees" type="number">
      Total fees charged for the transaction
    </ResponseField>
    
    <ResponseField name="recipient_details" type="object">
      Details of the recipient (for payouts)
    </ResponseField>
    
    <ResponseField name="sender_details" type="object">
      Details of the sender (for collections)
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the transaction was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp when the transaction was last updated
    </ResponseField>
    
    <ResponseField name="completed_at" type="string">
      Timestamp when the transaction was completed
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl --request GET \
  --url https://api-dev.blaaiz.com/api/external/transaction/txn_123456789 \
  --header 'x-blaaiz-api-key: YOUR_API_KEY'
```

```javascript Node.js
const response = await fetch('https://api-dev.blaaiz.com/api/external/transaction/txn_123456789', {
  method: 'GET',
  headers: {
    'x-blaaiz-api-key': 'YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api-dev.blaaiz.com/api/external/transaction/txn_123456789"
headers = {
    "x-blaaiz-api-key": "YOUR_API_KEY"
}

response = requests.get(url, headers=headers)
data = response.json()
```
</RequestExample>
