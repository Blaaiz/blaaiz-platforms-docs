---
title: Get a customer
description: Retrieve a single customer by ID
api: GET /api/external/customer/{id}
---

This endpoint allows you to retrieve the details of a single customer.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the customer
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The customer details
  
  <Expandable title="Customer object">
    <ResponseField name="id" type="string">
      Unique identifier for the customer
    </ResponseField>
    
    <ResponseField name="type" type="string">
      Customer type (individual or business)
    </ResponseField>
    
    <ResponseField name="first_name" type="string">
      Customer's first name
    </ResponseField>
    
    <ResponseField name="last_name" type="string">
      Customer's last name
    </ResponseField>
    
    <ResponseField name="business_name" type="string">
      Business name (if applicable)
    </ResponseField>
    
    <ResponseField name="email" type="string">
      Customer's email
    </ResponseField>
    
    <ResponseField name="phone_number" type="string">
      Customer's phone number
    </ResponseField>
    
    <ResponseField name="date_of_birth" type="string">
      Date of birth
    </ResponseField>
    
    <ResponseField name="address" type="object">
      Customer's address details
    </ResponseField>
    
    <ResponseField name="id_type" type="string">
      Type of identification document
    </ResponseField>
    
    <ResponseField name="id_number" type="string">
      Identification document number
    </ResponseField>
    
    <ResponseField name="files" type="array">
      Array of uploaded document files
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Customer status (pending, verified, rejected)
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the customer was created
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp when the customer was last updated
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl --request GET \
  --url https://api-dev.blaaiz.com/api/external/customer/customer_123456789 \
  --header 'x-blaaiz-api-key: YOUR_API_KEY'
```

```javascript Node.js
const response = await fetch('https://api-dev.blaaiz.com/api/external/customer/customer_123456789', {
  method: 'GET',
  headers: {
    'x-blaaiz-api-key': 'YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api-dev.blaaiz.com/api/external/customer/customer_123456789"
headers = {
    "x-blaaiz-api-key": "YOUR_API_KEY"
}

response = requests.get(url, headers=headers)
data = response.json()
```
</RequestExample>
