---
title: "Create a customer"
description: "Create a new customer record"
api: "POST /api/external/customer"
---

This endpoint allows you to create a new customer.

By default, this customer's verification status will be PENDING. You need to upload some files for the customer before the status moves to VERIFIED.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Body Parameters

<ParamField body="type" type="string" required>
  The type of customer. Options: `individual`, `business`
</ParamField>

<ParamField body="email" type="string" required>
  Customer's email address.
</ParamField>

<ParamField body="country" type="string" required>
  Country code (e.g., NG, CA, US).
</ParamField>

<ParamField body="id_type" type="string" required>
  Type of identification document. Options: `drivers_license`, `passport`, `id_card`, `resident_permit`, `certificate_of_incorporation`. The ID type can only be `certificate_of_incorporation` when the selected type is `business`, and it cannot be `certificate_of_incorporation` when the selected type is `individual`.
</ParamField>

<ParamField body="id_number" type="string" required>
  Identification document number.
</ParamField>

<ParamField body="first_name" type="string">
  Customer's first name. Required if type is `individual`.
</ParamField>

<ParamField body="last_name" type="string">
  Customer's last name. Required if type is `individual`.
</ParamField>

<ParamField body="business_name" type="string">
  Business name. Required when type is `business`.
</ParamField>

<ParamField body="phone" type="string">
  Customer's phone number.
</ParamField>

<ParamField body="tin" type="string">
  Tax identification number. The expected TIN type varies by country and can be retrieved from the [Get identification type](/api-reference/virtual-bank-account/get-identification-type) endpoint. For example, in Nigeria this would be the Bank Verification Number (BVN).
</ParamField>

<ParamField body="dob" type="string">
  Date of birth in YYYY-MM-DD format.
</ParamField>

<ParamField body="street" type="string">
  Street address.
</ParamField>

<ParamField body="city" type="string">
  City.
</ParamField>

<ParamField body="state" type="string">
  State or province.
</ParamField>

<ParamField body="zip_code" type="string">
  Postal or ZIP code.
</ParamField>

<ParamField body="id_expiry_date" type="string">
  Expiry date of the identification document in YYYY-MM-DD format.
</ParamField>

<ParamField body="id_issue_date" type="string">
  Issue date of the identification document in YYYY-MM-DD format.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Response message.
</ResponseField>

<ResponseField name="data" type="object">
  The created customer object.

  <Expandable title="Customer object">
    <ResponseField name="id" type="string">
      Unique identifier for the customer.
    </ResponseField>
    <ResponseField name="business_id" type="string">
      The business ID associated with this customer.
    </ResponseField>
    <ResponseField name="first_name" type="string">
      Customer's first name.
    </ResponseField>
    <ResponseField name="last_name" type="string">
      Customer's last name.
    </ResponseField>
    <ResponseField name="business_name" type="string">
      Business name (if applicable).
    </ResponseField>
    <ResponseField name="type" type="string">
      Customer type (individual or business).
    </ResponseField>
    <ResponseField name="email" type="string">
      Customer's email.
    </ResponseField>
    <ResponseField name="country" type="string">
      Customer's country code.
    </ResponseField>
    <ResponseField name="id_type" type="string">
      Type of identification document.
    </ResponseField>
    <ResponseField name="id_number" type="string">
      Identification document number.
    </ResponseField>
    <ResponseField name="verification_status" type="string">
      Customer verification status (PENDING, VERIFIED).
    </ResponseField>
  </Expandable>
</ResponseField>
