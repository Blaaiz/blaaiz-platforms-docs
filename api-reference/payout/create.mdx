---
title: Create a payout
description: Initiate a payout to a customer
api: POST /api/external/payout
---

Initiate a payout to send funds to a customer's bank account or via other supported methods.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Body Parameters

<ParamField body="customer_id" type="string" required>
  The unique identifier of the customer receiving the payout
</ParamField>

<ParamField body="wallet_id" type="string" required>
  The wallet ID from which funds will be debited
</ParamField>

<ParamField body="amount" type="number" required>
  The amount to send (in the smallest currency unit, e.g., kobo for NGN)
</ParamField>

<ParamField body="currency" type="string" required>
  The currency code (e.g., NGN, CAD, USD)
</ParamField>

<ParamField body="method" type="string" required>
  The payout method. Options: `bank_transfer`, `interac`
</ParamField>

<ParamField body="bank_code" type="string">
  The bank code (required for bank_transfer method)
</ParamField>

<ParamField body="account_number" type="string">
  The recipient's account number (required for bank_transfer method)
</ParamField>

<ParamField body="account_name" type="string">
  The recipient's account name (required for bank_transfer method)
</ParamField>

<ParamField body="email" type="string">
  The recipient's email (required for interac method)
</ParamField>

<ParamField body="reference" type="string">
  Your unique reference for this transaction
</ParamField>

<ParamField body="narration" type="string">
  Description or note for the payout
</ParamField>

<ParamField body="interac_question" type="string">
  Security question for Interac transfer (required for interac method)
</ParamField>

<ParamField body="interac_answer" type="string">
  Answer to the security question (required for interac method)
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The payout transaction details
  
  <Expandable title="Payout object">
    <ResponseField name="id" type="string">
      Unique identifier for the payout
    </ResponseField>
    
    <ResponseField name="transaction_id" type="string">
      The transaction ID
    </ResponseField>
    
    <ResponseField name="customer_id" type="string">
      The customer ID
    </ResponseField>
    
    <ResponseField name="wallet_id" type="string">
      The wallet ID
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      The payout amount
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency code
    </ResponseField>
    
    <ResponseField name="method" type="string">
      The payout method used
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status of the payout (pending, processing, completed, failed)
    </ResponseField>
    
    <ResponseField name="reference" type="string">
      Your unique reference
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the payout was created
    </ResponseField>
  </Expandable>
</ResponseField>
