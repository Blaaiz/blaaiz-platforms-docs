---
title: Retrieve a single virtual bank account
description: Retrieve a single virtual bank account by ID
api: GET /api/external/virtual-bank-account/{id}
---

Retrieve details of a specific virtual bank account using its unique identifier.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the virtual bank account
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The virtual bank account object
  
  <Expandable title="Virtual bank account object">
    <ResponseField name="id" type="string">
      Unique identifier for the virtual bank account
    </ResponseField>
    
    <ResponseField name="account_number" type="string">
      The virtual account number
    </ResponseField>
    
    <ResponseField name="account_name" type="string">
      The name on the virtual account
    </ResponseField>
    
    <ResponseField name="bank_name" type="string">
      The bank name for the virtual account
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency of the virtual account
    </ResponseField>
    
    <ResponseField name="wallet_id" type="string">
      The wallet ID associated with this virtual account
    </ResponseField>
    
    <ResponseField name="customer_id" type="string">
      The customer ID associated with this virtual account
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status of the virtual account
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the virtual account was created
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl --request GET \
  --url https://api-dev.blaaiz.com/api/external/virtual-bank-account/vba_123456789 \
  --header 'x-blaaiz-api-key: YOUR_API_KEY'
```

```javascript Node.js
const response = await fetch('https://api-dev.blaaiz.com/api/external/virtual-bank-account/vba_123456789', {
  method: 'GET',
  headers: {
    'x-blaaiz-api-key': 'YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

url = "https://api-dev.blaaiz.com/api/external/virtual-bank-account/vba_123456789"
headers = {
    "x-blaaiz-api-key": "YOUR_API_KEY"
}

response = requests.get(url, headers=headers)
data = response.json()
```
