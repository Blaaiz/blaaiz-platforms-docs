---
title: Create virtual bank account
description: Create a new virtual bank account for a customer
api: POST /api/external/virtual-bank-account
---

This endpoint allows businesses to generate a new virtual bank account (VBA) associated with a specific wallet. This account can be used for transactions in the designated wallet's currency.

**Currently, this supports NGN, GBP, USD and EUR.**

<Note>
All virtual accounts — except NGN — are only available to customers with **type = individual**.
</Note>

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Body Parameters

<ParamField body="wallet_id" type="string" required>
  The ID of the business wallet the account will be associated with.
</ParamField>

<ParamField body="account_name" type="string">
  The name for the virtual bank account holder, with a max length of 255 characters. Required if currency is NGN.
</ParamField>

<ParamField body="customer_id" type="string">
  The ID of the customer the VBA should be created for.
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The created virtual bank account object
  
  <Expandable title="Virtual bank account object">
    <ResponseField name="id" type="string">
      Unique identifier for the virtual bank account
    </ResponseField>
    
    <ResponseField name="account_number" type="string">
      The virtual account number
    </ResponseField>
    
    <ResponseField name="account_name" type="string">
      The name on the virtual account
    </ResponseField>
    
    <ResponseField name="bank_name" type="string">
      The bank name for the virtual account
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency of the virtual account
    </ResponseField>
    
    <ResponseField name="wallet_id" type="string">
      The wallet ID associated with this virtual account
    </ResponseField>
    
    <ResponseField name="customer_id" type="string">
      The customer ID associated with this virtual account
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status of the virtual account
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the virtual account was created
    </ResponseField>
  </Expandable>
</ResponseField>

  }'
