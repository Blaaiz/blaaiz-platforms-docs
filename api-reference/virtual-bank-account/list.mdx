---
title: Retrieve all virtual bank accounts
description: Retrieve all virtual bank accounts
api: GET /api/external/virtual-bank-account
---

This endpoint retrieves all virtual bank accounts (VBAs) associated with a specified wallet. By passing the `wallet_id` parameter, you can fetch all VBAs created under that particular wallet for the authenticated business.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Query Parameters

<ParamField query="wallet_id" type="string" required>
  The unique identifier of the wallet to which the virtual bank accounts are
  linked.
</ParamField>

<ParamField query="customer_id" type="string">
  The unique identifier of the customer you want to retrieve for (if provided).
</ParamField>

## Response

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="array">
  Array of virtual bank account objects
  <Expandable title="Virtual bank account object">
    <ResponseField name="id" type="string">
      Unique identifier for the virtual bank account
    </ResponseField>

    <ResponseField name="business_wallet_id" type="string">
      The business wallet ID associated with this virtual account
    </ResponseField>

    <ResponseField name="currency_id" type="number">
      Identifier of the currency for the virtual account
    </ResponseField>

    <ResponseField name="account_number" type="string">
      The virtual account number
    </ResponseField>

    <ResponseField name="iban" type="string | null">
      IBAN for the virtual account (if applicable)
    </ResponseField>

    <ResponseField name="account_name" type="string">
      The name on the virtual account
    </ResponseField>

    <ResponseField name="bank_name" type="string">
      The bank name for the virtual account
    </ResponseField>

    <ResponseField name="bank_code" type="string">
      The bank code for the virtual account
    </ResponseField>

    <ResponseField name="currency" type="string">
      The currency of the virtual account (e.g., NGN, USD)
    </ResponseField>

    <ResponseField name="customer_id" type="string">
      The customer ID associated with this virtual account
    </ResponseField>

    <ResponseField name="account_reference" type="string">
      Reference associated with the virtual account
    </ResponseField>

    <ResponseField name="reservation_reference" type="string | null">
      Reservation reference for the virtual account, if present
    </ResponseField>

    <ResponseField name="sort_code" type="string | null">
      Sort code for the virtual account, if applicable
    </ResponseField>

    <ResponseField name="bank_address" type="string | null">
      Bank address for the virtual account, if applicable
    </ResponseField>

    <ResponseField name="status" type="string">
      Status of the virtual account (e.g., SUCCESSFUL)
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Timestamp when the virtual account was created
    </ResponseField>

  </Expandable>
</ResponseField>
