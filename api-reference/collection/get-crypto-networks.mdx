---
title: Get crypto networks
description: Retrieve all available blockchain networks and their supported tokens.
api: GET /api/external/collection/crypto/networks
---

This endpoint allows you to fetch all available networks and their tokens.

## Endpoint

```
GET /api/external/collection/crypto/networks
```

## Headers

<ParamField header="Accept" type="string" default="application/json">
  Content type header.
</ParamField>

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Success message.
</ResponseField>

<ResponseField name="data" type="object">
  Network data object.
  
  <Expandable title="Data object">
    <ResponseField name="success" type="boolean">
      Whether the request was successful.
    </ResponseField>
    
    <ResponseField name="message" type="string">
      Success message.
    </ResponseField>
    
    <ResponseField name="status" type="number">
      HTTP status code.
    </ResponseField>
    
    <ResponseField name="data" type="array">
      Array of available networks.
      
      <Expandable title="Network object">
        <ResponseField name="id" type="number">
          Network ID.
        </ResponseField>
        
        <ResponseField name="name" type="string">
          Network name.
        </ResponseField>
        
        <ResponseField name="identifier" type="string">
          Network identifier (e.g., ETHEREUM_MAINNET).
        </ResponseField>
        
        <ResponseField name="tokens" type="array">
          Array of supported tokens.
          
          <Expandable title="Token object">
            <ResponseField name="id" type="number">
              Token ID.
            </ResponseField>
            
            <ResponseField name="name" type="string">
              Token name.
            </ResponseField>
            
            <ResponseField name="symbol" type="string">
              Token symbol (e.g., USDT, USDC).
            </ResponseField>
            
            <ResponseField name="address" type="string">
              Token contract address.
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
