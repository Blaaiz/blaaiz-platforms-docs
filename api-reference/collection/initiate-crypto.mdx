---
title: Initiate crypto collection
description: Initiate a crypto collection to fund your wallet using cryptocurrency.
api: POST /api/external/collection/crypto
---

This endpoint allows you to initiate a crypto collection, enabling you to fund your wallet using cryptocurrency.

## Endpoint

```
POST /api/external/collection/crypto
```

## Headers

<ParamField header="Accept" type="string" default="application/json">
  Content type header.
</ParamField>

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

## Body parameters

<ParamField body="amount" type="number" required>
  Amount to collect (minimum value of 0.1).
</ParamField>

<ParamField body="wallet_id" type="string" required>
  The ID of the wallet to fund.
</ParamField>

<ParamField body="network" type="string" required>
  The blockchain network to use (e.g., ETHEREUM_MAINNET, MATIC_MAINNET, TRON_MAINNET).
</ParamField>

<ParamField body="token" type="string" required>
  The token to use (e.g., USDT, USDC).
</ParamField>

<ParamField body="customer_id" type="string">
  Existing customer ID to associate with the collection.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Success message.
</ResponseField>

<ResponseField name="transaction" type="object">
  Transaction details.
  
  <Expandable title="Transaction object">
    <ResponseField name="transaction_id" type="string">
      Unique transaction identifier.
    </ResponseField>
    
    <ResponseField name="reference" type="string">
      Transaction reference.
    </ResponseField>
    
    <ResponseField name="token" type="string">
      Token used for the transaction.
    </ResponseField>
    
    <ResponseField name="token_amount" type="number">
      Amount in tokens.
    </ResponseField>
    
    <ResponseField name="network" type="string">
      Blockchain network used.
    </ResponseField>
    
    <ResponseField name="address" type="string">
      Wallet address to send funds to.
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Transaction status.
    </ResponseField>
    
    <ResponseField name="expires_at" type="string">
      Expiration timestamp.
    </ResponseField>
  </Expandable>
</ResponseField>
