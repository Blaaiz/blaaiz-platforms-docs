---
title: Initiate a collection
description: Create a collection request to receive payment from a customer
api: POST /api/external/collection
---

Initiate a collection to receive payment from a customer using various payment methods like card, open banking, or virtual accounts.

## Headers

<ParamField header="x-blaaiz-api-key" type="string" required>
  Your API key for authentication
</ParamField>

## Body Parameters

<ParamField body="customer_id" type="string" required>
  The unique identifier of the customer making the payment
</ParamField>

<ParamField body="wallet_id" type="string" required>
  The wallet ID where funds will be credited
</ParamField>

<ParamField body="amount" type="number" required>
  The amount to collect (including all fees)
</ParamField>

<ParamField body="currency" type="string" required>
  The currency code (e.g., NGN, USD, EUR, GBP)
</ParamField>

<ParamField body="method" type="string" required>
  The collection method. Options: `card`, `open_banking`, `virtual_account`
</ParamField>

<ParamField body="phone_number" type="string">
  Customer's phone number (required for certain methods)
</ParamField>

<ParamField body="email" type="string">
  Customer's email address
</ParamField>

<ParamField body="reference" type="string">
  Your unique reference for this transaction
</ParamField>

<ParamField body="narration" type="string">
  Description or note for the collection
</ParamField>

<ParamField body="redirect_url" type="string">
  URL to redirect the customer after payment completion
</ParamField>

## Response

<ResponseField name="status" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  The collection transaction details
  
  <Expandable title="Collection object">
    <ResponseField name="id" type="string">
      Unique identifier for the collection
    </ResponseField>
    
    <ResponseField name="transaction_id" type="string">
      The transaction ID
    </ResponseField>
    
    <ResponseField name="customer_id" type="string">
      The customer ID
    </ResponseField>
    
    <ResponseField name="wallet_id" type="string">
      The wallet ID
    </ResponseField>
    
    <ResponseField name="amount" type="number">
      The collection amount
    </ResponseField>
    
    <ResponseField name="currency" type="string">
      The currency code
    </ResponseField>
    
    <ResponseField name="method" type="string">
      The collection method used
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status of the collection (pending, processing, completed, failed)
    </ResponseField>
    
    <ResponseField name="payment_url" type="string">
      URL for the customer to complete payment (for card and open_banking methods)
    </ResponseField>
    
    <ResponseField name="reference" type="string">
      Your unique reference
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the collection was created
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL - Card Payment
curl --request POST \
  --url https://api-dev.blaaiz.com/api/external/collection \
  --header 'Content-Type: application/json' \
  --header 'x-blaaiz-api-key: YOUR_API_KEY' \
  --data '{
    "customer_id": "customer_123",
    "wallet_id": "wallet_123",
    "amount": 50000,
    "currency": "NGN",
    "method": "card",
    "phone_number": "+2348012345678",
    "email": "john@example.com",
    "reference": "collection_ref_001",
    "narration": "Payment for services",
    "redirect_url": "https://yourapp.com/payment/success"
  }'
```

```python Python
import requests

url = "https://api-dev.blaaiz.com/api/external/collection"
headers = {
    "Content-Type": "application/json",
    "x-blaaiz-api-key": "YOUR_API_KEY"
}
payload = {
    "customer_id": "customer_123",
    "wallet_id": "wallet_123",
    "amount": 50000,
    "currency": "NGN",
    "method": "card",
    "phone_number": "+2348012345678",
    "email": "john@example.com",
    "reference": "collection_ref_001",
    "narration": "Payment for services",
    "redirect_url": "https://yourapp.com/payment/success"
}

response = requests.post(url, json=payload, headers=headers)
data = response.json()
```
